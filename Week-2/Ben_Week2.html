fout <- vector()

    f <- sample(size = 107, dat$Family, replace = TRUE)
    out <- vector()
    for (i in f) {
        out <- c(out, dat[dat$Family %in% f, "Clutch_size"])
    }
    fout <- c(fout, mean(out))
}
